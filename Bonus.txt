DROP TABLE IF EXISTS navers CASCADE;
DROP TABLE IF EXISTS projects CASCADE;
DROP TABLE IF EXISTS projects_navers;


CREATE TABLE IF NOT EXISTS navers(
    id serial PRIMARY KEY,
    name VARCHAR(50),
    birthdate TIMESTAMP NOT NULL,
    admission_date TIMESTAMP NOT NULL,
    job_role VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_DATE,
    updated_at TIMESTAMP DEFAULT CURRENT_DATE
);

CREATE TABLE IF NOT EXISTS projects(
    id serial PRIMARY KEY,
    name VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_DATE,
    updated_at TIMESTAMP DEFAULT CURRENT_DATE
);

CREATE TABLE IF NOT EXISTS projects_navers(
    id serial PRIMARY KEY,
    project_id INTEGER NOT NULL,
    naver_id INTEGER NOT NULL,
    CONSTRAINT projects_navers_naver_id_foreign FOREIGN KEY (naver_id)
        REFERENCES navers(id) MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE CASCADE,
    CONSTRAINT projects_navers_project_id_foreign FOREIGN KEY (project_id)
        REFERENCES projects(id) MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE CASCADE
);
